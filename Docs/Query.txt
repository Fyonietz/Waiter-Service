-- 1. Table Roles
CREATE TABLE Roles(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT NOT NULL
);

-- 2. Table Users
CREATE TABLE Users(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT NOT NULL,
    Password TEXT NOT NULL,
    Role_Id INTEGER,
    FOREIGN KEY (Role_Id) REFERENCES Roles(Id)
);

-- 3. Table Types
CREATE TABLE Types(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT NOT NULL
);

-- 4. Table Status
CREATE TABLE Status(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT NOT NULL
);

-- 5. Menu
CREATE TABLE Menu(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT,
    Price REAL,
    Type_Id INTEGER,
    FOREIGN KEY (Type_Id) REFERENCES Types(Id)
);

-- 6. Location
CREATE TABLE Location(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT
);

-- 7. Orders_On_Waiter
CREATE TABLE Orders_On_Waiter(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Id_User INTEGER,
    Customer_Name TEXT NOT NULL,
    Date DATETIME DEFAULT CURRENT_TIMESTAMP,
    Id_Status INTEGER,
    Id_Location INTEGER,
    FOREIGN KEY (Id_User) REFERENCES Users(Id),
    FOREIGN KEY (Id_Status) REFERENCES Status(Id),
    FOREIGN KEY (Id_Location) REFERENCES Location(Id)
);

-- 8. Orders_Items
CREATE TABLE Orders_Items(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Id_Order INTEGER,
    Id_Menu INTEGER,
    Quantity INTEGER NOT NULL,
    Price_At_Order REAL,
    FOREIGN KEY (Id_Order) REFERENCES Orders_On_Waiter(Id),
    FOREIGN KEY (Id_Menu) REFERENCES Menu(Id)
);

-- 9. Orders_On_Chef
CREATE TABLE Orders_On_Chef(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Id_Order INTEGER,
    Id_User INTEGER,
    FOREIGN KEY (Id_Order) REFERENCES Orders_On_Waiter(Id),
    FOREIGN KEY (Id_User) REFERENCES Users(Id)
);

-- 10. Orders_On_Cashier
CREATE TABLE Orders_On_Cashier(
    Id INTEGER PRIMARY KEY AUTOINCREMENT,
    Id_Order INTEGER,
    Id_User INTEGER,
    FOREIGN KEY (Id_Order) REFERENCES Orders_On_Waiter(Id),
    FOREIGN KEY (Id_User) REFERENCES Users(Id)
);
